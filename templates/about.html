Technical Architecture


class StoryForgeAI:
    def __init__(self, tone, genre, topic):
        self.tone = tone
        self.genre = genre
        self.topic = topic
        self.story_generator = AIStoryGenerator()
    
    def generate_story_outline(self):
        context = self._prepare_context()
        outline = self.story_generator.generate(context)
        return outline
    
    def _prepare_context(self):
        return {
            'tone': self.tone,
            'genre': self.genre,
            'topic': self.topic,
            'complexity': 'medium'
        }
